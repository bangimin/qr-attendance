function doGet(e) { if (!e || !e.parameter) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: 'Parameter tidak ditemukan' })) .setMimeType(ContentService.MimeType.JSON); } // Endpoint khusus statistik if (e.parameter.action === 'statistik') { return ContentService .createTextOutput(JSON.stringify(getStatistik())) .setMimeType(ContentService.MimeType.JSON); } if (!e.parameter.qr) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: 'Parameter QR tidak ditemukan' })) .setMimeType(ContentService.MimeType.JSON); } const result = checkin(e.parameter.qr); return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getStatistik() { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName('peserta'); const data = sheet.getDataRange().getValues(); const total = data.length - 1; // minus header let hadir = 0; for (let i = 1; i < data.length; i++) { if (data[i][2] === 'HADIR') hadir++; } return { hadir: hadir, total: total }; } function checkin(qr) { const lock = LockService.getScriptLock(); lock.waitLock(5000); try { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName('peserta'); if (!sheet) { return { success: false, message: 'Sheet "peserta" tidak ditemukan' }; } const data = sheet.getDataRange().getValues(); const total = data.length - 1; let hadir = 0; for (let i = 1; i < data.length; i++) { if (data[i][2] === 'HADIR') hadir++; } for (let i = 1; i < data.length; i++) { const row = data[i]; const qrCode = row[0]; const nama = row[1]; const status = row[2]; if (String(qrCode).trim() === String(qr).trim()) { if (status === 'HADIR') { return { success: false, message: 'SUDAH CHECK-IN', nama: nama, hadir: hadir, total: total }; } sheet.getRange(i + 1, 3).setValue('HADIR'); sheet.getRange(i + 1, 4).setValue(new Date()); return { success: true, message: 'VALID', nama: nama, hadir: hadir + 1, total: total }; } } return { success: false, message: 'QR TIDAK DITEMUKAN', hadir: hadir, total: total }; } catch(err) { return { success: false, message: 'Error: ' + err.message }; } finally { lock.releaseLock(); } }
